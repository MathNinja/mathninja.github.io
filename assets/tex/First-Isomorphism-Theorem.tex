\textbf{THE FIRST ISOMORPHISM THEOREM:}

\[
\begin{tikzcd}[column sep=5em,row sep=5em]
    G \arrow{r}{\phi} \arrow{d}[left]{\pi} & \text{Im}(\phi) \subseteq H \\
    G/\text{Ker}(\phi) \arrow[dashrightarrow]{ru}[below right]{\mu\,(\cong)} 
    %& \text{Im}(\phi) \arrow{u}[right]{\subseteq (\text{includes})} 
\end{tikzcd}
\]
 
\textbf{CONCRETE EXAMPLE:}
  
\[
\phi: \mathbb{Z}_{6} \to \mathbb{Z}_{4}, \quad \textrm{where} \quad
\phi(x) = 2x \: \textrm{mod}\:4
\]  
 

\[ 
\begin{tikzcd}[column sep=5em,row sep=5em]
    \mathbb{Z}_{6} \arrow{r}{\phi} \arrow{d}[left]{\pi} & \{ 0,\,2 \} \subseteq \mathbb{Z}_4 \\
    \mathbb{Z}_{6}/\left\{ 0, 2, 4 \right\} \arrow[dashrightarrow]{ru}[below right]{\mu\,(\cong)} 
    %& \text{Im}(\phi) \arrow{u}[right]{\subseteq (\text{includes})} 
\end{tikzcd}
\]
