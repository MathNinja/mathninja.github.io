\textbf{THE THIRD ISOMORPHISM THEOREM:}

\[
\begin{tikzcd}[column sep=5em,row sep=5em]
    G \arrow{r}{\gamma_H} \arrow{d}[left]{\gamma_K} & G/H \\
    G/K \arrow{r}{\gamma_{H/K}} & (G/K)/(H/K) \arrow[leftrightarrow]{u}[right]{\mu\,(\cong)} 
\end{tikzcd}
\]
 
\textbf{CONCRETE EXAMPLE:}

\begin{align*}
& H = \{0,2,4,6\} \\[3pt]
& K = \{0,4\} \\[3pt]
& H/K = \big\{ \{0,4\}, \{2,6\} \big\}
\end{align*}


\begin{align*}
& \gamma_H: \mathbb{Z}_{8} \to \mathbb{Z}_{8}/H, \quad \textrm{where} \quad \gamma_H(x) = x + H \\[3pt]
& \gamma_K: \mathbb{Z}_{8} \to \mathbb{Z}_{8}/K, \quad \textrm{where} \quad \gamma_K(x) = x + K \\[3pt]
& \gamma_H/K: \mathbb{Z}_{8}/K \to (\mathbb{Z}_{8}/K)/(\mathbb{Z}_{8}/H), \quad \textrm{where} \quad \gamma_{H/K}(x) = x + H/K
\end{align*}

\[
\begin{tikzcd}[column sep=5em,row sep=5em]
    \mathbb{Z}_{8} \arrow{r}{\gamma_H} \arrow{d}[left]{\gamma_K} & \mathbb{Z}_{8}/\{0,2,4,6\} \\
    \mathbb{Z}_{8}/\{0,4\} \arrow{r}{\gamma_{H/K}} & (\mathbb{Z}_{8}/\{0,4\})/(\mathbb{Z}_{8}/\{0,2,4,6\}) \arrow[leftrightarrow]{u}[right]{\mu\,(\cong)} 
\end{tikzcd}
\]
