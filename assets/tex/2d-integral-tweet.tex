% #1 #2 #3 dimesions

\newcommand{\myColumn}[8]{
	% Bottom rectangle
	\plot3[solid, #4, line width=\axwidth] coordinates {(#1+#6,#2+#7,0+#8) (#1+#6,0+#7,0+#8) (0+#6,0+#7,0+#8) (0+#6,#2+#7,0+#8) (#1+#6,#2+#7,0+#8)};
	% Back edge
	\plot3[solid, #4, line width=\axwidth] 
		coordinates {(0+#6,0+#7,0+#8) (0+#6,0+#7,#3+#8)};
	% Fill front side
	\plot3[draw=none, fill=#4!30, fill opacity=#5] 
		coordinates {(#1+#6,0+#7,0+#8) (#1+#6,#2+#7,0+#8) (#1+#6,#2+#7,#3+#8) (#1+#6,0+#7,#3+#8)};	
	% Fill top side
	\plot3[draw=none, fill=#4!30, fill opacity=#5] 
		coordinates {(0+#6,0+#7,#3+#8) (#1+#6,0+#7,#3+#8) (#1+#6,#2+#7,#3+#8) (0+#6,#2+#7,#3+#8)};	
	% Fill right side
	\plot3[draw=none, fill=#4!30, fill opacity=#5] 
		coordinates {(0+#6,#2+#7,0+#8) (#1+#6,#2+#7,0+#8) (#1+#6,#2+#7,#3+#8) (0+#6,#2+#7,#3+#8)};
	% Edges		
	\plot3[solid, #4, line width=\axwidth] 
		coordinates {(#1+#6,#2+#7,0+#8) (#1+#6,#2+#7,#3+#8)};
	\plot3[solid, #4, line width=\axwidth] 
		coordinates {(#1+#6,0+#7,0+#8) (#1+#6,0+#7,#3+#8)};
	\plot3[solid, #4, line width=\axwidth] 
		coordinates {(0+#6,#2+#7,0+#8) (0+#6,#2+#7,#3+#8)};
	% Top rectangle	
	\plot3[solid, #4, line width=\axwidth] coordinates {(#1+#6,#2+#7,#3+#8) (#1+#6,0+#7,#3+#8) (0+#6,0+#7,#3+#8) (0+#6,#2+#7,#3+#8) (#1+#6,#2+#7,#3+#8)};
	
}
